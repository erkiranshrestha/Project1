# Configuration: 

- git config --global user.email "your email"
- git config --global user.name "your name"

# To add file to Staging area: 

- git add file_name.txt
- git add file1.txt file2.txt => for two files
- git add . => for all files

# To commit the changes: 

- git commit -m "Message"


# To check the history: 

- git log => View commit history
- git log --oneline => view in oneline

# Move to previous commit location: 

- git checkout master => for moving to the main branch
- git checkout "SHA-1" => encrypted key


# Connect the file to your github: 

- copy remote origin from the github directory, then push it to the origin master:
- git push origin master
